define yum::plugin (
  String $content,
  String $plugin  = $title,
) {

  if ! defined(Class['::yum']) {
    fail('You must include the yum base class before using any yum defined resources')
  }

  file { "${::yum::pluginconf_dir}/${plugin}.conf":
    ensure  => file,
    owner   => 0,
    group   => 0,
    mode    => '0644',
    content => $content,
  }
}
