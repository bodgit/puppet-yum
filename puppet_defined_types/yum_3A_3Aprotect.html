define yum::protect (
  Array[String, 1] $packages = [$title],
) {

  if ! defined(Class['::yum']) {
    fail('You must include the yum base class before using any yum defined resources')
  }

  file { "${::yum::protected_dir}/${title}.conf":
    ensure  => file,
    owner   => 0,
    group   => 0,
    mode    => '0644',
    content => "${join($packages, "\n")}\n",
  }
}
